# Arc Setup Guide

Get started with Arc in under 5 minutes.

## Prerequisites

- Node.js 22+
- MongoDB (local or Atlas)
- npm or pnpm

## Quick Start

### 1. Install Dependencies

```bash
# Core (qs and fastify-plugin are bundled with Arc — no separate install needed)
npm install @classytic/arc fastify

# Database adapter (choose one)
npm install @classytic/mongokit mongoose    # MongoDB (recommended)
# Or bring your own adapter implementing Arc's DataAdapter interface

# Authentication (choose one)
npm install @fastify/jwt                    # Arc JWT auth
# npm install better-auth                  # Better Auth (recommended for SaaS)

# Optional: Security & performance
npm install @fastify/cors @fastify/helmet @fastify/rate-limit @fastify/under-pressure

# Dev dependencies
npm install -D typescript @types/node vitest pino-pretty
```

### 2. Initialize TypeScript

```bash
npx tsc --init
```

Update `tsconfig.json`:

```json
{
  "compilerOptions": {
    "target": "ES2022",
    "module": "NodeNext",
    "moduleResolution": "NodeNext",
    "esModuleInterop": true,
    "strict": true,
    "outDir": "dist",
    "rootDir": "src",
    "declaration": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist"]
}
```

### 4. Create Your First Resource

**src/modules/product/product.model.ts**
```typescript
import mongoose from 'mongoose';

const productSchema = new mongoose.Schema(
  {
    name: { type: String, required: true, trim: true },
    slug: { type: String, unique: true, sparse: true },
    price: { type: Number, required: true, min: 0 },
    description: { type: String },
    isActive: { type: Boolean, default: true },
    deletedAt: { type: Date, default: null },
  },
  { timestamps: true }
);

productSchema.index({ slug: 1 });
productSchema.index({ isActive: 1, deletedAt: 1 });

export const Product = mongoose.model('Product', productSchema);
```

**src/modules/product/product.repository.ts**
```typescript
import { Repository, softDeletePlugin } from '@classytic/mongokit';
import { Product } from './product.model.js';

class ProductRepository extends Repository<typeof Product> {
  constructor() {
    super(Product, [softDeletePlugin()]);
  }

  async findActive() {
    return this.Model.find({ isActive: true, deletedAt: null }).lean();
  }

  async getBySlug(slug: string) {
    return this.Model.findOne({ slug, deletedAt: null }).lean();
  }
}

export const productRepository = new ProductRepository();
```

**src/modules/product/product.resource.ts**
```typescript
import { defineResource, createMongooseAdapter, allowPublic, requireRoles } from '@classytic/arc';
import { Product } from './product.model.js';
import { productRepository } from './product.repository.js';
import { productController } from './product.controller.js';

export const productResource = defineResource({
  name: 'product',
  displayName: 'Products',
  prefix: '/products',

  adapter: createMongooseAdapter({
    model: Product,
    repository: productRepository,
  }),
  controller: productController,

  presets: ['softDelete', 'slugLookup'],

  permissions: {
    list: allowPublic(),
    get: allowPublic(),
    create: requireRoles(['admin']),
    update: requireRoles(['admin']),
    delete: requireRoles(['admin']),
    deleted: requireRoles(['admin']),  // softDelete preset
    restore: requireRoles(['admin']),  // softDelete preset
    getBySlug: allowPublic(),          // slugLookup preset
  },
});
```

### 5. Create App Entry Point

**src/index.ts**
```typescript
import mongoose from 'mongoose';
import { createApp } from '@classytic/arc/factory';
import { productResource } from './modules/product/product.resource.js';

async function bootstrap() {
  // Connect to MongoDB
  await mongoose.connect(process.env.MONGO_URI || 'mongodb://localhost:27017/my-api');
  console.log('Connected to MongoDB');

  // Create Arc app
  const app = await createApp({
    preset: 'development', // or 'production'
    auth: {
      type: 'jwt',
      jwt: { secret: process.env.JWT_SECRET || 'dev-secret-change-in-production' },
    },
    plugins: async (fastify) => {
      // Register resources
      await fastify.register(productResource.toPlugin());
    },
  });

  // Start server
  const port = parseInt(process.env.PORT || '3000');
  await app.listen({ port, host: '0.0.0.0' });
  console.log(`Server running at http://localhost:${port}`);

  // Graceful shutdown
  const shutdown = async () => {
    console.log('Shutting down...');
    await app.close();
    await mongoose.disconnect();
    process.exit(0);
  };

  process.on('SIGTERM', shutdown);
  process.on('SIGINT', shutdown);
}

bootstrap().catch((err) => {
  console.error('Failed to start:', err);
  process.exit(1);
});
```

### 6. Add Scripts to package.json

```json
{
  "type": "module",
  "scripts": {
    "dev": "tsx watch src/index.ts",
    "build": "tsc",
    "start": "node dist/index.js",
    "generate": "npx @classytic/arc generate"
  }
}
```

### 7. Run the App

```bash
# Development (with hot reload)
npm run dev

# Production
npm run build
npm start
```

### 8. Test Your API

```bash
# List products
curl http://localhost:3000/products

# Create product (requires auth in production)
curl -X POST http://localhost:3000/products \
  -H "Content-Type: application/json" \
  -d '{"name": "Widget", "price": 29.99}'

# Get by ID
curl http://localhost:3000/products/<id>

# Get by slug
curl http://localhost:3000/products/slug/widget

# Update
curl -X PATCH http://localhost:3000/products/<id> \
  -H "Content-Type: application/json" \
  -d '{"price": 24.99}'

# Delete (soft)
curl -X DELETE http://localhost:3000/products/<id>

# Health check
curl http://localhost:3000/_health/live
```

## Using the CLI

### Initialize a New Project

```bash
# Scaffold a complete Arc project
npx @classytic/arc init my-api --mongokit --better-auth --ts

# With options
npx @classytic/arc init my-api --mongokit --multi-tenant --ts --better-auth
npx @classytic/arc init my-api --custom --single --js --jwt --skip-install
```

### Generate Resources

```bash
# Generate full resource (model, repository, controller, routes, tests)
npx @classytic/arc generate resource order --module sales

# With presets
npx @classytic/arc generate resource invoice -p softDelete,multiTenant

# Preview without creating files
npx @classytic/arc generate resource payment --dry-run

# Generate specific files only
npx @classytic/arc generate controller auth
npx @classytic/arc generate model customer
```

### Introspect & Diagnostics

```bash
# Show all registered resources from an entry file
npx @classytic/arc introspect --entry ./dist/index.js

# Output JSON metadata for resources (AI agent friendly)
npx @classytic/arc describe ./dist/index.js

# Export OpenAPI specification
npx @classytic/arc docs ./openapi.json --entry ./dist/index.js

# Health check environment and dependencies
npx @classytic/arc doctor
```

## Production Setup

### Environment Files (Next.js-style)

Arc CLI projects load env files with this priority (first loaded wins):

| Priority | File | Purpose | Git |
|----------|------|---------|-----|
| 1 | `.env.local` | Machine-specific overrides (secrets, local DB) | **gitignored** |
| 2 | `.env.production` / `.env.prod` | Environment-specific defaults | committed |
| 2 | `.env.development` / `.env.dev` | Development defaults | committed |
| 2 | `.env.test` | Test environment defaults | committed |
| 3 | `.env` | Shared defaults (fallback) | committed |

All matching files are loaded and merged — earlier files take priority via `dotenv({ override: false })`.

**How it works:**
1. `NODE_ENV` is normalized: `production`/`prod` → `prod`, `development` → `dev`, `test`/`qa` → `test`
2. Both long-form (`.env.production`) and short-form (`.env.prod`) are checked
3. `.env.local` always loads first regardless of `NODE_ENV`
4. Already-set env vars (from shell, Docker, CI) are never overwritten

```bash
# .env (shared defaults — safe to commit)
PORT=8040
HOST=0.0.0.0
CORS_ORIGINS=http://localhost:3000

# .env.local (your machine — gitignored, NEVER committed)
DATABASE_URI=mongodb://localhost:27017/mydb
JWT_SECRET=your-32-char-minimum-secret-key
BETTER_AUTH_SECRET=your-secret-here

# .env.production (production defaults — committed)
NODE_ENV=production
PORT=3000

# .env.test (test defaults — committed)
NODE_ENV=test
PORT=0
```

> **Important:** The env loader (`src/config/env.ts`) must be the **first import** in your
> entry point — before any code that reads `process.env`. Arc CLI scaffolds this automatically.

### Environment Best Practices

| Do | Don't |
|----|-------|
| Put secrets in `.env.local` (gitignored) | Commit secrets in `.env` or `.env.production` |
| Use `.env.example` as a template for new devs | Rely on defaults for production secrets |
| Set `NODE_ENV` explicitly in CI/Docker | Leave `NODE_ENV` unset in production |
| Use `config.isProd` / `config.isDev` guards | Compare `process.env.NODE_ENV` directly |
| Validate required vars at startup | Silently fall back to empty strings for critical config |

**Typed config pattern** (generated by `arc init`):

```typescript
// src/config/index.ts — all env access goes through this file
export interface AppConfig {
  env: string;
  isDev: boolean;
  isProd: boolean;
  server: { port: number; host: string };
  database: { uri: string };
  // ... add your app-specific config here
}

const config: AppConfig = {
  env: process.env.NODE_ENV || 'development',
  isDev: (process.env.NODE_ENV || 'development') !== 'production',
  isProd: process.env.NODE_ENV === 'production',
  server: {
    port: parseInt(process.env.PORT || '8040', 10),
    host: process.env.HOST || '0.0.0.0',
  },
  database: {
    uri: process.env.DATABASE_URI || '',
  },
};

export default config;
```

> **Note:** Arc is database-agnostic. The config above uses `DATABASE_URI` as a generic
> name — your app decides what goes in it (MongoDB, PostgreSQL, MySQL, SQLite, etc.).

### Production App Configuration

```typescript
import { createApp } from '@classytic/arc/factory';

const app = await createApp({
  preset: 'production',
```

## Multi-Tenant Setup

For SaaS applications with organization isolation:

```typescript
import { defineResource, createMongooseAdapter, requireAuth } from '@classytic/arc';

export const invoiceResource = defineResource({
  name: 'invoice',
  prefix: '/invoices',

  adapter: createMongooseAdapter({
    model: Invoice,
    repository: invoiceRepository,
  }),

  // Enable multi-tenant isolation
  presets: ['multiTenant', 'softDelete'],
  organizationScoped: true,

  permissions: {
    list: requireAuth(),
    get: requireAuth(),
    create: requireAuth(),
    update: requireAuth(),
    delete: requireAuth(),
  },
});
```

Requests require `x-organization-id` header:

```bash
curl http://localhost:3000/invoices \
  -H "Authorization: Bearer <token>" \
  -H "x-organization-id: org_123"
```

## Adding Custom Routes

```typescript
export const productResource = defineResource({
  name: 'product',
  // ... adapter, permissions

  routes: [
    {
      method: 'GET',
      path: '/featured',
      handler: 'getFeatured',
      summary: 'Get featured products',
      permissions: allowPublic(),
    },
    {
      method: 'POST',
      path: '/:id/publish',
      handler: 'publish',
      summary: 'Publish a product',
      permissions: requireRoles(['admin']),
    },
  ],
});
```

Implement handlers in your controller:

```typescript
import { BaseController } from '@classytic/arc';
import type { IRequestContext, IControllerResponse } from '@classytic/arc';

class ProductController extends BaseController {
  async getFeatured(req: IRequestContext): Promise<IControllerResponse> {
    const products = await this.repository.Model.find({
      isFeatured: true,
      isActive: true
    }).limit(10);
    return { success: true, data: products };
  }

  async publish(req: IRequestContext): Promise<IControllerResponse> {
    const { id } = req.params;
    const product = await this.repository.update(id, {
      status: 'published',
      publishedAt: new Date(),
    });
    return { success: true, data: product };
  }
}
```

## Testing

**tests/product.test.ts**
```typescript
import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { createTestApp } from '@classytic/arc/testing';
import type { FastifyInstance } from 'fastify';

describe('Product API', () => {
  let app: FastifyInstance;

  beforeAll(async () => {
    app = await createTestApp({
      auth: { type: 'jwt', jwt: { secret: 'test-secret-32-chars-minimum-len' } },
    });
  });

  afterAll(async () => {
    await app?.close();
  });

  it('should list products', async () => {
    const response = await app.inject({
      method: 'GET',
      url: '/products',
    });

    expect(response.statusCode).toBe(200);
    expect(response.json().success).toBe(true);
  });

  it('should create a product', async () => {
    const response = await app.inject({
      method: 'POST',
      url: '/products',
      payload: { name: 'Test Product', price: 99.99 },
    });

    expect(response.statusCode).toBe(201);
    expect(response.json().data.name).toBe('Test Product');
  });
});
```

## Troubleshooting

### "Cannot find module" errors

Ensure your `package.json` has `"type": "module"` and imports use `.js` extensions:

```typescript
// Correct
import { Product } from './product.model.js';

// Wrong
import { Product } from './product.model';
```

### MongoDB connection issues

Check your connection string and ensure MongoDB is running:

```bash
# Local MongoDB
mongod --dbpath /data/db

# Or use Docker
docker run -d -p 27017:27017 mongo
```

### TypeScript path issues

If using path aliases, configure both `tsconfig.json` and your runtime:

```json
// tsconfig.json
{
  "compilerOptions": {
    "paths": {
      "#modules/*": ["./src/modules/*"]
    }
  }
}
```

```json
// package.json
{
  "imports": {
    "#modules/*": "./dist/modules/*"
  }
}
```
